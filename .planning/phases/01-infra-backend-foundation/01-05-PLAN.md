**`backend/alembic/env.py`** — CRITICAL: render_as_batch=True and compare_type=True MUST be present:
```python
"""
Alembic migration environment for Diet Coach Agent backend.

CRITICAL SQLite safeguards (BE-03, D-001):
  - render_as_batch=True: Required for SQLite DDL operations (ALTER TABLE, etc.)
    SQLite does not support ALTER COLUMN natively; batch mode recreates tables.
    Without this, all future schema changes will fail on SQLite.
  - compare_type=True: Detects column type changes during autogenerate.
    Without this, type changes are silently ignored.

These TWO flags MUST remain in context.configure() forever.
They are safe for PostgreSQL migration (ignored when using PostgreSQL).
"""
import os
import sys
from logging.config import fileConfig

from alembic import context
from sqlalchemy import engine_from_config, pool

# Ensure the backend/app package is importable from this directory
sys.path.insert(0, os.path.dirname(os.path.dirname(os.path.abspath(__file__))))

# Import settings to get DATABASE_URL
from app.core.config import settings  # noqa: E402

# Import Base (DeclarativeBase with all table metadata)
from app.core.database import Base  # noqa: E402

# ── CRITICAL: Import all models so Base.metadata is populated ────────────
# This single import triggers __init__.py which imports all 22 models.
# Alembic autogenerate ONLY sees tables registered with Base.metadata.
# If any model module is not imported here, its table will be missing.
import app.models  # noqa: F401, E402

# Alembic Config object
config = context.config

# Interpret the config file for Python logging.
if config.config_file_name is not None:
    fileConfig(config.config_file_name)

# Set target metadata for autogenerate support
target_metadata = Base.metadata

# Override sqlalchemy.url from pydantic settings (not from alembic.ini string)
config.set_main_option("sqlalchemy.url", settings.DATABASE_URL)


def run_migrations_offline() -> None:
    """
    Run migrations in 'offline' mode.
    Configures the context with just a URL and not an Engine.
    """
    url = config.get_main_option("sqlalchemy.url")
    context.configure(
        url=url,
        target_metadata=target_metadata,
        literal_binds=True,
        dialect_opts={"paramstyle": "named"},
        # ── MANDATORY SQLite safeguards ──────────────────────────────
        render_as_batch=True,   # Required for SQLite DDL (BE-03)
        compare_type=True,      # Detect column type changes
        # ─────────────────────────────────────────────────────────────
    )

    with context.begin_transaction():
        context.run_migrations()


def run_migrations_online() -> None:
    """
    Run migrations in 'online' mode.
    Creates an Engine and associates a connection with the context.
    """
    connectable = engine_from_config(
        config.get_section(config.config_ini_section, {}),
        prefix="sqlalchemy.",
        poolclass=pool.NullPool,
    )

    with connectable.connect() as connection:
        context.configure(
            connection=connection,
            target_metadata=target_metadata,
            # ── MANDATORY SQLite safeguards ──────────────────────────────
            render_as_batch=True,   # Required for SQLite DDL (BE-03)
            compare_type=True,      # Detect column type changes
            # ─────────────────────────────────────────────────────────────
        )

        with context.begin_transaction():
            context.run_migrations()


if context.is_offline_mode():
    run_migrations_offline()
else:
    run_migrations_online()
```
